Toast Management

// Toast management (title is required)
// Show multiple toasts
for (let i = 1; i <= 3; i++) {
    GenieAI.Toast.info(`Toast ${i}`, { title: `Info ${i}` });
}

// Update existing toast
const toast = GenieAI.Toast.show({
    id: 'updateable-toast',
    title: 'Initial Toast',
    message: 'Initial message',
    type: 'info'
});

setTimeout(() => {
    GenieAI.Toast.update('updateable-toast', {
        title: 'Updated Toast',
        message: 'Updated message!',
        type: 'success'
    });
}, 2000);

// Management methods
GenieAI.Toast.closeAll();
GenieAI.Toast.clear('error'); // Clear only error toasts
const count = GenieAI.Toast.getCount();
