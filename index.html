Toast Management

// Toast management (title is required)
// Show multiple toasts
for (let i = 1; i <= 3; i++) {
    GenieAIToast.info(`Toast ${i}`, { title: `Info ${i}` });
}

// Update existing toast
const toast = GenieAIToast.show({
    id: 'updateable-toast',
    title: 'Initial Toast',
    message: 'Initial message',
    type: 'info'
});

setTimeout(() => {
    GenieAIToast.update('updateable-toast', {
        title: 'Updated Toast',
        message: 'Updated message!',
        type: 'success'
    });
}, 2000);

// Management methods
GenieAIToast.closeAll();
GenieAIToast.clear('error'); // Clear only error toasts
const count = GenieAIToast.getCount();
